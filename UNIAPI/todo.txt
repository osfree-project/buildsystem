x. add default modifier to uses

uses os2 default;

will produce

#define INCL_OS2
#ifdef INCL_OS2
#include <os2.h>
#endif

x. add alias modifier to uses

uses bse alias base;

will produce

#ifdef INCL_BASE
#define INCL_BSE
#endif
#ifdef INCL_BSE
#include <bse.h>
#endif
